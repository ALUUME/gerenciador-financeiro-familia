import csv from datetime import datetime # Função para exibir o menu principal def exibir_menu(): print("\n--- Gerenciador de Gastos Familiares ---") print("1. Adicionar gasto") print("2. Exibir gastos") print("3. Sair") # Função para adicionar um gasto def adicionar_gasto(): nome = input("Nome do membro da família (Pai, Mãe, Juliana, Maria): ").capitalize() if nome not in ["Pai", "Mãe", "Juliana", "Maria"]: print("Nome inválido. Tente novamente.") return descricao = input("Descrição do gasto: ") valor = float(input("Valor do gasto: ")) data = datetime.now().strftime("%Y-%m-%d %H:%M:%S") with open('gastos.csv', 'a', newline='') as file: writer = csv.writer(file) writer.writerow([nome, descricao, valor, data]) print("Gasto adicionado com sucesso!") # Função para exibir todos os gastos def exibir_gastos(): try: with open('gastos.csv', 'r') as file: reader = csv.reader(file) print("\n--- Gastos Registrados ---") for row in reader: print(f"Membro: {row[0]}, Descrição: {row[1]}, Valor: R${row[2]:.2f}, Data: {row[3]}") except FileNotFoundError: print("Nenhum gasto registrado ainda.") # Função principal para executar o programa def main(): while True: exibir_menu() escolha = input("Escolha uma opção: ") if escolha == '1': adicionar_gasto() elif escolha == '2': exibir_gastos() elif escolha == '3': print("Saindo do programa...") break else: print("Opção inválida. Tente novamente.") if __name__ == "__main__": main()
